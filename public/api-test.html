📱 Expo 앱 연결 테스트

위의 API 테스트가 모두 성공하면 Expo 앱에서 다음 코드를 사용할 수 있습니다:

// 기본 API 상태 확인 fetch('http://172.16.29.142:8000/api/', { method: 'GET', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json', } }) .then(response => response.json()) .then(data => console.log('API 상태:', data)) .catch(error => console.error('에러:', error)); // 공개 카테고리 조회 fetch('http://172.16.29.142:8000/api/v1/categories/public', { method: 'GET', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json', } }) .then(response => response.json()) .then(data => console.log('카테고리:', data)) .catch(error => console.error('에러:', error)); // 간단한 로그인 fetch('http://172.16.29.142:8000/api/debug/simple-login', { method: 'POST', headers: { 'Content-Type': 'application/json', 'Accept': 'application/json', }, body: JSON.stringify({ email: 'parent@test.com', password: 'password123' }) }) .then(response => response.json()) .then(data => console.log('로그인 결과:', data)) .catch(error => console.error('에러:', error));
